  • new operator
new operator
function Foo(){
    this.bar = function(){ console.log('bar'); }
}
var myFoo = new Foo();
myFoo.bar();
  1. A new object is created linked to Foo.prototype
  2. Foo Function is called with this set to the new object
  3. The new operator returns this object or if provided, the result of the Foo Function

  • Prototype
Prototype
function Foo(){
    this.bar = function(){ console.log('bar'); }
}
Foo.prototype.blap = function(){ console.log('blap'); }
var myFoo = new Foo();
myFoo.bar();
myFoo.blap();
Foo.prototype.boink = function(){ console.log('boink'); }
myFoo.boink();
  1. Once the function is defined the prototype can be accessed
  2. The prototype instance is shared and can be extended after the new call
  • delete operator
delete operator
function Foo(){
    this.bar = function(){ console.log('bar'); }
}
Foo.prototype.blap = function(){ console.log('blap'); }

var myFoo = new Foo();
myFoo.bar();

delete myFoo.bar
console.log('bar deleted');

myFoo.bar(); //->TypeError: Object #Foo has no method 'bar'

var myFoo2 = new Foo();
myFoo2.blap();

delete Foo.prototype.blap
console.log('blap deleted');

myFoo2.blap(); // TypeError: Object #
Foo has no method 'blap'
myFoo2.bar(); // Still works
myFoo.blap(); // Doesn't work either - TypeError: Object #
Foo has no method 'blap'
  1. delete can be used to remove attributes and functions
  2. Can be used on both the prototype and object instances
